### 1. Wind API数据获取标准流程

#### 1.1 数据加载函数结构
```python
def _load_xxx_data() -> pd.DataFrame:
    """加载XXX数据 - 数据库优先，Wind API兜底"""
    # DB优先
    try:
        df = db_manager.load_data('table_name')
        if df is not None and not df.empty:
            # 1. 检查date列是否存在
            if 'date' in df.columns:
                df['date'] = pd.to_datetime(df['date'], errors='coerce')
                # 2. 处理列名映射（value -> 实际列名）
                df = df[['date', 'value']].rename(columns={'value': 'actual_column_name'})
                # 3. 数值转换
                df['actual_column_name'] = pd.to_numeric(df['actual_column_name'], errors='coerce')
                # 4. 清理和排序
                df = df.dropna().sort_values('date')
                # 5. 确保日期为月初（月度数据）
                df['date'] = _ensure_month_start(df['date'])
                return df
    except Exception:
        pass
    
    # Wind API兜底
    try:
        import WindPy
        w = WindPy.w
        
        # 启动Wind API（如果未连接）
        try:
            if not w.isconnected():
                w.start()
            if not w.isconnected():
                return pd.DataFrame(columns=['date', 'column_name'])
        except Exception as e:
            return pd.DataFrame(columns=['date', 'column_name'])
        
        # 获取10年历史数据
        end_date = pd.Timestamp.now().strftime('%Y-%m-%d')
        start_date = (pd.Timestamp.now() - pd.DateOffset(years=10)).strftime('%Y-%m-%d')
        
        # 调用Wind API
        data = w.edb("CODE1,CODE2", start_date, end_date, "Fill=Previous")
        
        # 检查返回结果
        if not hasattr(data, 'ErrorCode'):
            return pd.DataFrame(columns=['date', 'column_name'])
        
        if data.ErrorCode != 0:
            return pd.DataFrame(columns=['date', 'column_name'])
        
        if not data.Data or len(data.Data) == 0 or not data.Times:
            return pd.DataFrame(columns=['date', 'column_name'])
        
        # 处理数据
        dates = pd.to_datetime(data.Times)
        values = pd.Series(data.Data[0])  # 单系列用[0]，多系列用[0], [1]等
        df = pd.DataFrame({'date': dates, 'column_name': values})
        df['column_name'] = pd.to_numeric(df['column_name'], errors='coerce')
        df = df.dropna().sort_values('date')
        df['date'] = _ensure_month_start(df['date'])
        
        # 保存到数据库（列名映射为value/value2等）
        try:
            db_manager.save_data(df.rename(columns={'column_name': 'value'}), 
                                'table_name', if_exists='replace')
        except Exception:
            pass
        
        return df
    except Exception as e:
        return pd.DataFrame(columns=['date', 'column_name'])
```

### 2. 日期处理关键点

#### 2.1 _ensure_month_start函数
```python
def _ensure_month_start(s: pd.Series) -> pd.Series:
    """确保日期为月初"""
    s = pd.to_datetime(s, errors='coerce')
    return s.dt.to_period('M').dt.start_time
```

#### 2.2 数据库读取时的日期解析
- 使用`pd.to_datetime(df['date'], errors='coerce')`进行转换
- 如果失败，尝试多种格式：直接转换、字符串转换、infer_datetime_format

### 3. 数据库列名映射规则

- **单系列数据**：数据库列名 `value` → 代码中列名 `actual_name`
- **多系列数据**：数据库列名 `value`, `value2` → 代码中列名 `series1`, `series2`
- **读取时**：需要检查列名并正确映射

## 二、图表生成流程

### 1. 单折线图标准结构
```python
def get_xxx_chart() -> go.Figure:
    """获取XXX图表"""
    data = _load_xxx_data()
    
    if data.empty:
        return None
    
    # 取最近10年
    end_date = data['date'].max()
    start_date = end_date - pd.DateOffset(years=10)
    data = data[(data['date'] >= start_date) & (data['date'] <= end_date)]
    
    if data.empty:
        return None
    
    fig = go.Figure()
    
    # 添加折线
    fig.add_trace(go.Scatter(
        x=data['date'], y=data['value'],
        name='指标名称', line=dict(color='#3498db', width=2),
        connectgaps=False
    ))
    
    # 添加零线（如果需要）
    fig.add_hline(y=0, line_dash="dash", line_color="gray", opacity=0.5)
    
    # 计算Y轴范围
    value_range = data['value'].dropna()
    pad = (value_range.max() - value_range.min()) * 0.05 if not value_range.empty else 1
    
    fig.update_layout(
        title='图表标题',
        xaxis=dict(title='日期'),
        yaxis=dict(title='Y轴标签', range=[value_range.min()-pad, value_range.max()+pad] if not value_range.empty else None, zeroline=True),
        template='plotly_white',
        height=400,
        margin=dict(l=50, r=50, t=50, b=50),
        legend=dict(x=0.02, y=0.98)
    )
    return fig
```

```python
def get_xxx_yyy_chart() -> go.Figure:
    """获取XXX vs YYY对比图表（双折线图）"""
    data = _load_xxx_yyy_data()
    
    if data.empty:
        return None
    
    # 取最近10年
    end_date = data['date'].max()
    start_date = end_date - pd.DateOffset(years=10)
    data = data[(data['date'] >= start_date) & (data['date'] <= end_date)]
    
    if data.empty:
        return None
    
    fig = go.Figure()
    
    # 第一个系列（左Y轴）
    fig.add_trace(go.Scatter(
        x=data['date'], y=data['series1'],
        name='系列1名称', line=dict(color='#3498db', width=2),
        connectgaps=False, yaxis='y'
    ))
    
    # 第二个系列（右Y轴）
    fig.add_trace(go.Scatter(
        x=data['date'], y=data['series2'],
        name='系列2名称', line=dict(color='#e74c3c', width=2),
        connectgaps=False, yaxis='y2'
    ))
    
    # 计算Y轴范围
    range1 = data['series1'].dropna()
    range2 = data['series2'].dropna()
    
    pad1 = (range1.max() - range1.min()) * 0.05 if not range1.empty else 1
    pad2 = (range2.max() - range2.min()) * 0.05 if not range2.empty else 1
    
    fig.update_layout(
        title='图表标题',
        xaxis=dict(title='日期'),
        yaxis=dict(title='左Y轴标签', range=[range1.min()-pad1, range1.max()+pad1] if not range1.empty else None),
        yaxis2=dict(title='右Y轴标签', side='right', overlaying='y', range=[range2.min()-pad2, range2.max()+pad2] if not range2.empty else None),
        template='plotly_white',
        height=400,
        margin=dict(l=50, r=50, t=50, b=50),
        legend=dict(x=0.02, y=0.98)
    )
    return fig
```

## 四、常见问题和解决方案

### 1. 数据提取问题

#### 问题1：数据库日期范围不完整
- **原因**：数据库保存时日期转换问题，或数据更新不及时
- **解决**：检查`_ensure_month_start`函数是否正确应用，确保数据库保存时日期格式一致

#### 问题2：多系列数据列名映射错误
- **原因**：数据库列名（value/value2）与代码列名不一致
- **解决**：在`_load_xxx_data()`函数中正确映射列名，支持多种列名格式

#### 问题3：图表数据点少于原始数据
- **原因**：10年过滤导致数据减少，或数据合并时丢失
- **解决**：检查日期过滤逻辑，确保数据合并时使用`how='outer'`保留所有数据

### 2. 图表显示问题

#### 问题1：图表不显示
- **原因**：函数返回None，或数据为空
- **解决**：添加空数据检查，确保函数始终返回有效的Figure对象或None

#### 问题2：Y轴范围不合理
- **原因**：数据范围计算错误，或padding设置不当
- **解决**：使用5%的padding，检查数据范围计算逻辑

#### 问题3：双Y轴重叠
- **原因**：Y轴范围设置不当
- **解决**：分别为两个Y轴设置独立的范围，使用padding避免重叠

### 3. Wind API问题

#### 问题1：Wind API连接失败
- **原因**：Wind终端未启动或未登录
- **解决**：添加详细的错误提示，指导用户启动Wind终端

#### 问题2：数据代码错误
- **原因**：Wind代码不存在或已变更
- **解决**：先测试Wind API接口，确认代码可用后再集成

#### 问题3：多系列数据获取失败
- **原因**：`data.Data`长度不足，或系列顺序错误
- **解决**：检查`len(data.Data)`，确保有足够的系列数据

## 六、代码规范

### 1. 函数命名
- 数据加载函数：`_load_xxx_data()`
- 图表生成函数：`get_xxx_chart()`
- 页面显示函数：`show_xxx_page()`

### 2. 颜色规范
- 主系列：`#3498db`（蓝色）
- 次系列：`#e74c3c`（红色）
- 第三系列：`#2ecc71`（绿色）

- 标准高度：400px
- 边距：`dict(l=50, r=50, t=50, b=50)`
- 图例位置：`dict(x=0.02, y=0.98)`

### 4. 错误处理
- 所有函数都应包含try-except
- 返回空DataFrame而不是None（数据加载函数）
- 返回None而不是空Figure（图表生成函数）
- 提供清晰的错误提示
